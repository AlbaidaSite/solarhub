### User ###
    username = models.CharField(max_length=150, unique=True)
    name = models.CharField(max_length=255)
    email = models.EmailField(unique=True, validators=[EmailValidator()])
    password = models.CharField(max_length=128)
    profile_img = models.ImageField(
        upload_to='static/media/img/profile_images/',
        default=static('media/img/default_profile.png')
    )
    is_active = models.BooleanField(default=False)
    is_staff = models.BooleanField(default=False)
    is_superuser = models.BooleanField(default=False)
    is_loukou = models.BooleanField(default=False)


### Cromo ### 
    name = models.CharField(max_length=100, default=None)
    front_img = models.URLField(default='https://i.imgur.com/BpGTZ9D.png')
    back_img = models.URLField(default='https://i.imgur.com/5oh8HlY.png')
    description = models.TextField(default='Default description')
    number = models.IntegerField(default=None)
    copies = models.IntegerField(default=1)
    rarity = models.ForeignKey('Rarity', default='Común', on_delete=models.CASCADE, related_name='cromos')
    category = models.ForeignKey('Category', default='Neutral', on_delete=models.CASCADE, related_name='cromos')
    artists = models.ManyToManyField('Artist', related_name='cromos')
    is_owned_by_user = models.BooleanField(default=False)
    has_owners = models.BooleanField(default=False)
    how_to = models.TextField(default='Aparece en sobres')
    how_to_extended = models.TextField(default='Aparece de forma aleatoria en sobres.')
    hide_til_registered = models.BooleanField(default=False)
    for_loukous = models.BooleanField(default=True)
    allow_multiple_users = models.BooleanField(default=False)  # Define si puede pertenecer a múltiples usuarios


### Unique ###
    code = models.IntegerField(default=None, validators=[
            MinValueValidator(1),
            MaxValueValidator(65535)
        ])
    cromo = models.ForeignKey('Cromo', on_delete=models.CASCADE, related_name='uniques', default=None)
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='uniques', default=None, null=True)
    number = models.IntegerField(default=1)


### UniqueOwnership ###
    cromo = models.ForeignKey(Cromo, on_delete=models.CASCADE, related_name='ownerships', null=True)
    user = models.ForeignKey(User, on_delete=models.CASCADE, related_name='owned_cromos', null=True)
    date_acquired = models.DateTimeField(auto_now_add=True, null=True)  # Fecha en que el usuario obtuvo el cromo
    is_current_owner = models.BooleanField(default=True)
